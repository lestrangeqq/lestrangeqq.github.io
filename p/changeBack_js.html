Изменение цвета бэкграунда по клику

HTML:

<button id='button' onclick='changeBack()'>Click</button>

CSS:

button {
  background-color:red;
  color:white;
  font-size:18px;
  padding:15px;
  border: 1px solid white;
}

JS:

let colors = ['red', 'blue', 'yellow']; //массив с цветами для бэкграунда
let i = 0; //переменная нужна чтобы бэкграунд динамично менялся на цвета из массивов. 0 - red, и т.д.

function changeBack() {
	i++; //каждый раз кликая по кнопке и запуская функцию к переменной будет прибавляться 1, то есть если переменная была i=0; i++ прибавит к ней 1 и выведет другой цвет из массива.
  if (i == 3) i = 0; //т.к. у нас 3 элемента в массиве [0, 1, 2], то по достижению i==3 мы сбрасываем переменную i опять до 0.
	document.querySelector('body').style.background = colors[i]; //colors[i] - выведет элемент (номер цвета) из массива равный текущему значению переменной i.
}
